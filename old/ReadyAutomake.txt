# prefix

#================================================
# Autoconf
#================================================
# download the archive containing the latest autoconf source 
wget http://ftp.gnu.org/gnu/autoconf/autoconf-2.59.tar.gz

# extract its contents
tar zxvf autoconf-2.59.tar.gz

# configure script
./configure

# build autoconf
make

# execute the tests provided with autoconf
make check

# install it
make install

#================================================
# Automake
#================================================
# download the archive containing the latest automake source

# extract the contents 
tar zxvf automake-1.9.5.tar.gz

# configure script 
./configure

# run the tests provided with Automake
make check
