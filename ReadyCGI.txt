yum install httpd
yum install php
yum install php-cli
yum install mod_fcgid

nano /etc/httpd/conf.d/php.conf
#<FilesMatch \.php$>
#    SetHandler application/x-httpd-php
#</FilesMatch>

updatedb
locate php.ini
locate php-cgi

nano /var/www/cgi-bin/php.fastcgi
#!/bin/bash

PHPRC="/etc/php.ini"
PHP_FCGI_CHILDREN=4
PHP_FCGI_MAX_REQUESTS=1000
export PHPRC
export PHP_FCGI_CHILDREN
export PHP_FCGI_MAX_REQUESTS
exec /usr/bin/php-cgi

ll /var/www/cgi-bin/php.fastcgi
-rw-r--r--. 1 root root 171 Jul  5 03:56 /var/www/cgi-bin/php.fastcgi

chown apache:apache /var/www/cgi-bin/php.fastcgi
chmod +x /var/www/cgi-bin/php.fastcgi

ll /var/www/cgi-bin/php.fastcgi
-rwxr-xr-x. 1 apache apache 171 Jul  5 03:56 /var/www/cgi-bin/php.fastcgi

nano /etc/httpd/conf.d/virtualhosts.conf
nano /etc/httpd/conf/httpd.conf

<VirtualHost *:80>
    ServerName www.readydev.com
    ServerAdmin admin@readydev.com
    DocumentRoot /var/www/html
    ScriptAlias /cgi-bin/ "/var/www/cgi-bin/"
    
    <Directory "/var/www/html">
        AddHandler php-fastcgi .php
        Action php-fastcgi /cgi-bin/php.fastcgi
        Options All
        AllowOverride All
        Order allow,deny
        Allow from All
        Require all granted
    </Directory>
</VirtualHost>

service httpd enable
service httpd restart
